 10class GameManager:
 11    """
 12    Manages the core game state and logic, including interactions.
 13    """
 14    def __init__(self, board):
 15        """
 16        Initialize the game manager with a specific board state.
 17        """
 18        self.board = board
 19        self.history = Stack()
 20        self.game_over = False
 21        self.game_won = False
 22        self.solver = MinesweeperSolver()
 23
 24    def reset_game(self, board):
 25        """
 26        Reset the game manager with a new board state.
 27        """
 28        self.board = board
 29        self.history = Stack()
 30        self.game_over = False
 31        self.game_won = False
 32        self.solver = MinesweeperSolver()
 33
 34    def save_state(self):
 35        """
 36        Save the current board state to the history stack for undo functionality.
 37        """
 38        state = copy.deepcopy(self.board.grid)
 39        self.history.push(state)
 40
 41    def undo(self):
 42        """
 43        Revert the board to its previous state.
 44        """
 45        if not self.history.is_empty():
 46            previous_state = self.history.pop()
 47            self.board.grid = previous_state
 48            self.game_over = False
 49            self.game_won = False
 50            
 51    def handle_left_click(self, x, y):
 52        """
 53        Handle a left-click event (reveal a cell).
 54        """
 55        if self.game_over or self.game_won:
 56            return
 57            
 58        cell = self.board.grid[x][y]
 59        
 60        if cell.is_flagged or cell.is_revealed:
 61            return
 62            
 63        self.save_state()
 64            
 65        if not self.board.mines_placed:
 66            self.board.place_mines(x, y)
 67            
 68        if cell.is_mine:
 69            cell.is_revealed = True
 70            self.game_over = True
 71            self.reveal_all_mines()
 72        else:
 73            if cell.adjacent_mines == 0:
 74                flood_fill_bfs(self.board.grid, x, y)
 75            else:
 76                cell.is_revealed = True
 77                
 78        self.check_win()
 79
 80    def handle_right_click(self, x, y):
 81        """
 82        Handle a right-click event (toggle flag).
 83        """
 84        if self.game_over or self.game_won:
 85            return
 86            
 87        cell = self.board.grid[x][y]
 88        if not cell.is_revealed:
 89            self.save_state()
 90            cell.is_flagged = not cell.is_flagged
 91
 92    def reveal_all_mines(self):
 93        """
 94        Reveal all mines on the board (typically after game over).
 95        """
 96        for r in range(self.board.rows):
 97            for c in range(self.board.cols):
 98                if self.board.grid[r][c].is_mine:
 99                    self.board.grid[r][c].is_revealed = True
100
101    def check_win(self):
102        """
103        Check if the player has won the game by revealing all safe cells.
104        """
105        for r in range(self.board.rows):
106            for c in range(self.board.cols):
107                cell = self.board.grid[r][c]
108                if not cell.is_mine and not cell.is_revealed:
109                    return False
110        self.game_won = True
111
112    def get_analysis(self):
113        """
114        Activates MinesweeperSolver to analyze the current board.
115        Returns a dictionary categorizing the cells:
116        - certain_mines: List of coordinates that are definitely mines.
117        - certain_safe: List of coordinates that are definitely safe.
118        - others: List of other cell coordinates sorted from lowest to highest mine probability.
119        """
120        if self.game_over or self.game_won:
121            return None
122            
123        probabilities = self.solver.execute_solve(self.board)
124        
125        certain_mines = []
126        certain_safe = []
127        others = []
128        
129        for (x, y), prob in probabilities.items():
130            if prob >= 0.999:
131                certain_mines.append((x, y))
132            elif prob <= 0.001:
133                certain_safe.append((x, y))
134            else:
135                others.append(((x, y), prob))
136                
137        others.sort(key=lambda item: item[1])
138        
139        return {
140            'certain_mines': certain_mines,
141            'certain_safe': certain_safe,
142            'others': [item[0] for item in others]
143        }
144
145    def auto_solve_step(self):
146        """
147        Executes a single step of the auto-solver (performing only one action to simulate a player).
148        Returns True if an action was taken, False otherwise.
149        """
150        analysis = self.get_analysis()
151        if not analysis:
152            return False
153            
154        for x, y in analysis['certain_mines']:
155            cell = self.board.grid[x][y]
156            if not cell.is_flagged and not cell.is_revealed:
157                self.handle_right_click(x, y)
158                return True
159                
160        for x, y in analysis['certain_safe']:
161            cell = self.board.grid[x][y]
162            if not cell.is_flagged and not cell.is_revealed:
163                self.handle_left_click(x, y)
164                return True
165                
166        if analysis['others']:
167            x, y = analysis['others'][0]
168            self.handle_left_click(x, y)
169            action_taken = True
170            
171        return False
