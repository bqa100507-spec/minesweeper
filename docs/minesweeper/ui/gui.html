 50    def create_menu(self):
 51        """
 52        Create the application menu bar including game difficulty options.
 53        """
 54        menubar = tk.Menu(self.root)
 55        self.root.config(menu=menubar)
 56        
 57        game_menu = tk.Menu(menubar, tearoff=0)
 58        menubar.add_cascade(label="Game", menu=game_menu)
 59        game_menu.add_command(label="New Game (Easy)", command=lambda: self.new_game(9, 9, 10))
 60        game_menu.add_command(label="New Game (Medium)", command=lambda: self.new_game(16, 16, 40))
 61        game_menu.add_command(label="New Game (Hard)", command=lambda: self.new_game(16, 30, 99))
 62        game_menu.add_command(label="Custom...", command=self.show_custom_dialog)
 63        game_menu.add_separator()
 64        game_menu.add_command(label="Exit", command=self.root.quit)

 66    def create_grid_widgets(self):
 67        """
 68        Create and layout the board buttons.
 69        """
 70        for r in range(self.board.rows):
 71            row_buttons = []
 72            for c in range(self.board.cols):
 73                btn = tk.Button(self.grid_frame, width=3, height=1, font=('Arial', 10, 'bold'))
 74                btn.bind("<Button-1>", lambda e, x=r, y=c: self.on_left_click(x, y))
 75                btn.bind("<Button-3>", lambda e, x=r, y=c: self.on_right_click(x, y))
 76                btn.grid(row=r, column=c)
 77                row_buttons.append(btn)
 78            self.buttons.append(row_buttons)

 80    def new_game(self, rows, cols, num_mines):
 81        """
 82        Start a new game with the specified board dimensions and mine count.
 83        """

 85        if self.timer_id:

101    def show_custom_dialog(self):
102        """
103        Show a dialog allowing the user to enter custom board dimensions and mine count.
104        """
105        dialog = tk.Toplevel(self.root)
106        dialog.title("Custom Game")
107        
108        tk.Label(dialog, text="Rows:").grid(row=0, column=0, padx=5, pady=5)
109        rows_entry = tk.Entry(dialog)
110        rows_entry.insert(0, str(self.board.rows))
111        rows_entry.grid(row=0, column=1, padx=5, pady=5)
112        
113        tk.Label(dialog, text="Columns:").grid(row=1, column=0, padx=5, pady=5)
114        cols_entry = tk.Entry(dialog)

116        cols_entry.grid(row=1, column=1, padx=5, pady=5)

121        mines_entry.grid(row=2, column=1, padx=5, pady=5)
122        
123        def on_ok():
124            try:
125                r = int(rows_entry.get())
126                c = int(cols_entry.get())
127                m = int(mines_entry.get())
128                if r < 1 or c < 1 or m < 1 or m >= r * c:
129                    messagebox.showerror("Error", "Invalid inputs. Ensure mines < rows * cols.", parent=dialog)
130                    return
131                self.new_game(r, c, m)
132                dialog.destroy()
133            except ValueError:
134                messagebox.showerror("Error", "Please enter valid integers.", parent=dialog)
135                
136        tk.Button(dialog, text="OK", command=on_ok).grid(row=3, column=0, columnspan=2, pady=10)

138    def hint_action(self):
139        """
140        Provide a hint for the next best move by highlighting a cell.
141        Safe cells are highlighted green, mines are red, and best guesses are yellow.
142        """
143        if self.game_manager.game_over or self.game_manager.game_won:
144            return
145            
146        if not self.board.mines_placed:
147            messagebox.showinfo("Hint", "Start by clicking any cell to generate the board.")
148            return
149
150        analysis = self.game_manager.get_analysis()
151        if not analysis:
152            messagebox.showinfo("Hint", "Unable to determine a sure move right now.")
153            return
154            
155        for x, y in analysis['certain_safe']:
156            cell = self.board.grid[x][y]
157            if not cell.is_flagged and not cell.is_revealed:
158                self.buttons[x][y].config(bg="lightgreen")
159                return
160                
161        for x, y in analysis['certain_mines']:
162            cell = self.board.grid[x][y]
163            if not cell.is_flagged and not cell.is_revealed:
164                self.buttons[x][y].config(bg="lightcoral")
165                return
166                
167        if analysis['others']:
168            x, y = analysis['others'][0]
169            self.buttons[x][y].config(bg="lightyellow")
170            messagebox.showinfo("Hint", f"No certain moves. The safest guess is cell ({x}, {y}).")
171            
172    def check_timer_state(self):
173        """
174        Check the current game state and start or stop the timer accordingly.
175        """
176        if self.board.mines_placed and not self.game_manager.game_over and not self.game_manager.game_won:
177            if not self.timer_running:
178                self.timer_running = True
179                self.update_timer()
180        else:
181            if self.timer_running:
182                self.timer_running = False
183                if self.timer_id:
184                    self.root.after_cancel(self.timer_id)
185                    self.timer_id = None

187    def update_timer(self):
188        """
189        Update the timer label with the elapsed seconds.
190        """
191        if self.timer_running:
192            self.timer_seconds += 1
193            self.timer_label.config(text=f"Time: {self.timer_seconds:03d}")
194            self.timer_id = self.root.after(1000, self.update_timer)

196    def update_gui(self):
197        """
198        Update the visual state of all buttons on the board to reflect the current game state.
199        """
200        for r in range(self.board.rows):
201            for c in range(self.board.cols):
202                cell = self.board.grid[r][c]
203                btn = self.buttons[r][c]
204                
205                if cell.is_revealed:
206                    btn.config(relief=tk.SUNKEN, state=tk.DISABLED, bg="#d3d3d3")
207                    if cell.is_mine:
208                        btn.config(text="*", disabledforeground="red")
209                    elif cell.adjacent_mines > 0:
210                        colors = ["", "blue", "green", "red", "purple", "maroon", "turquoise", "black", "gray"]
211                        color = colors[cell.adjacent_mines] if cell.adjacent_mines < len(colors) else "black"
212                        btn.config(text=str(cell.adjacent_mines), disabledforeground=color)
213                    else:
214                        btn.config(text="")
215                else:
216                    if btn.cget("bg") not in ["lightgreen", "lightcoral", "lightyellow"]:
217                        btn.config(state=tk.NORMAL, relief=tk.RAISED, bg="SystemButtonFace")
218                    else:
219                        btn.config(state=tk.NORMAL, relief=tk.RAISED)
220                        
221                    if cell.is_flagged:
222                        btn.config(text="F", fg="red")
223                    else:
224                        btn.config(text="")
225                        
226        self.check_timer_state()
227                        
228        if self.game_manager.game_over:
229            messagebox.showinfo("Game Over", "You hit a mine!\nPress Undo to try again.")
230        elif self.game_manager.game_won:
231            messagebox.showinfo("Congratulations", f"You won in {self.timer_seconds} seconds!")
232

273        """
274        Executes one auto-solve step, then reschedules itself after 600ms for a slower animation effect.
275        """
276        if self.game_manager.game_over or self.game_manager.game_won:
277            return

279        action_taken = self.game_manager.auto_solve_step()

150

152        if not analysis:
153            messagebox.showinfo("Hint", "Unable to determine a sure move right now.")
154            return
155            
156        for x, y in analysis['certain_safe']:
157            cell = self.board.grid[x][y]
158            if not cell.is_flagged and not cell.is_revealed:
159                self.buttons[x][y].config(bg="lightgreen")
160                return
161                
162        for x, y in analysis['certain_mines']:
163            cell = self.board.grid[x][y]
164            if not cell.is_flagged and not cell.is_revealed:
165                self.buttons[x][y].config(bg="lightcoral")
166                return
167                
168        if analysis['others']:
169            x, y = analysis['others'][0]
170            self.buttons[x][y].config(bg="lightyellow")
171            messagebox.showinfo("Hint", f"No certain moves. The safest guess is cell ({x}, {y}).")
